# OpenGL 창 생성 예제

이 예제는 OpenGL과 GLFW를 사용하여 간단한 창을 생성하고, 배경 색상을 설정하는 기본적인 프로그램입니다. 초보자를 위해 상세한 주석과 함께 작성되었습니다.

---

## 코드

```cpp
#include <iostream>
#include <glad/glad.h>
#include <GLFW/glfw3.h>

int main()
{
    // GLFW 라이브러리 초기화
    if (!glfwInit())
    {
        std::cout << "Failed to initialize GLFW" << std::endl;
        return -1; // 초기화 실패 시 프로그램 종료
    }

    // OpenGL 버전 설정 (3.3 버전 사용)
    glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3); // OpenGL 메이저 버전 3
    glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3); // OpenGL 마이너 버전 3
    glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE); // 코어 프로파일 사용

    // 800x800 크기의 창 생성 및 이름 설정
    GLFWwindow* window = glfwCreateWindow(800, 800, "YoutubeOpenGL", NULL, NULL);
    if (window == NULL)
    {
        std::cout << "Failed to create GLFW window" << std::endl;
        glfwTerminate(); // 창 생성 실패 시 GLFW 종료
        return -1;
    }

    // OpenGL 컨텍스트를 생성한 창과 연결
    glfwMakeContextCurrent(window);

    // GLAD를 통해 OpenGL 함수 로드
    if (!gladLoadGL())
    {
        std::cout << "Failed to initialize GLAD" << std::endl;
        return -1; // GLAD 초기화 실패 시 프로그램 종료
    }

    // OpenGL 뷰포트 설정 (화면에 그래픽을 그릴 영역 지정)
    // 여기서는 (0, 0)에서 (800, 800)까지 창 전체를 사용
    glViewport(0, 0, 800, 800);

    // 배경 색상을 설정 (짙은 회청색)
    glClearColor(0.07f, 0.13f, 0.17f, 1.0f);

    // 초기 화면 색상을 클리어 (배경 색으로 초기화)
    glClear(GL_COLOR_BUFFER_BIT);

    // 백 버퍼와 프론트 버퍼를 교체 (화면에 배경 색상 출력)
    glfwSwapBuffers(window);

    // 메인 루프: 창이 닫히기 전까지 계속 실행
    while (!glfwWindowShouldClose(window))
    {
        // 이벤트 처리 (키보드, 마우스 등)
        glfwPollEvents();
    }

    // 프로그램 종료 전에 생성한 창 삭제
    glfwDestroyWindow(window);

    // GLFW 리소스 해제
    glfwTerminate();

    return 0; // 프로그램 정상 종료
}
